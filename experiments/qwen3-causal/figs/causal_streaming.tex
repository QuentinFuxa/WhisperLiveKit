% ---------- row 1: mel blocks over time ----------
\foreach \i/\name in {0/B$_{k-4}$,1/B$_{k-3}$,2/B$_{k-2}$,3/B$_{k-1}$}{

\node[blk, draw=drop, dashed, fill=white, text=drop] (bdrop) at (-1.95, 0) {B$_{k-5}$};
% current block
\node[blk, draw=accent, very thick, fill=accentlight] (bcur) at (4*1.95, 0) {\textbf{B$_k$}};
% future audio
\node[blk, draw=drop, dashed, fill=white, text=inkgray] (bnext) at (5*1.95, 0) {$\cdots$};
\draw[-{Stealth[length=2mm]}, inkgray] ($(bnext.east)+(0.12,0)$) -- ++(0.85,0)
  node[lbl, right=1pt] {audio (1.92\,s blocks)};

% intra-block bidirectional attention
\draw[{Stealth[length=2.1mm]}-{Stealth[length=2.1mm]}, accent, line width=1.0pt]
  ($(bcur.south west)+(0.16,0.17)$) -- ($(bcur.south east)+(-0.16,0.17)$);

  {bidirectional\\\emph{within} the block};

% causal attention to frozen per-layer KV
\foreach \i in {1,2,3}{
  \draw[attn] (bcur.north) .. controls ($(bcur.north)+(-1.2,0.62)$)
    and ($(b\i.north)+(1.0,0.62)$) .. ($(b\i.north)+(0.1,0.05)$);
}
\node[lbl, accent] at (2.0, 1.30) {causal $\rightarrow$ frozen KV cache};

% sliding window bracket
\draw[decorate, decoration={brace, mirror, amplitude=4pt}, inkgray]
  ($(b0.south west)+(0,-0.16)$) -- ($(bcur.south east)+(0,-0.16)$)
  node[midway, below=5pt, lbl] {sliding attention window (15\,s) — older KV evicted};

\node[lbl, text=frozenedge, align=center] at (bdrop.north) [above=1pt] {evicted};
