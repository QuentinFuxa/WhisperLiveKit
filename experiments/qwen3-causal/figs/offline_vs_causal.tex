\newcommand{\waveform}[3]{% x0, length, opacity
  \begin{scope}[opacity=#3]
  \foreach \i in {0,...,#2}{
    \pgfmathsetmacro{\h}{0.10 + 0.16*abs(sin(\i*47) * sin(\i*13+40)) + 0.10*abs(sin(\i*89))}
    \draw[line width=0.55pt, inkgray] ({#1 + \i*0.066}, -\h) -- ({#1 + \i*0.066}, \h);
  }
  \end{scope}
}
